apiVersion: extensions/v1beta1
kind: Ingress
metadata:
  name: external-prox
  annotations:
    kubernetes.io/ingress.class: nginx
    nginx.ingress.kubernetes.io/backend-protocol: "HTTPS"
    nginx.ingress.kubernetes.io/auth-type: "basic"
    nginx.ingress.kubernetes.io/auth-secret: "nginx-basic-auth-jeff"
    #nginx.ingress.kubernetes.io/preserve-host: "true"
spec:
  rules:
  - host: prox.$DOMAIN
    http:
      paths:
      - path: /
        backend:
            serviceName: external-prox
            servicePort: 8006
      - path: /
        backend:
            serviceName: external-prox-b
            servicePort: 8006
      - path: /
        backend:
            serviceName: external-prox-c
            servicePort: 8006
      - path: "/vncwebsocket.*"
        backend:
            serviceName: external-prox-wss
            servicePort: 8006
      - path: "/vncwebsocket.*"
        backend:
            serviceName: external-prox-b-wss
            servicePort: 8006
      - path: "/vncwebsocket.*"
        backend:
            serviceName: external-prox-c-wss
            servicePort: 8006
  tls:
  - hosts:
    - prox.$DOMAIN
---
